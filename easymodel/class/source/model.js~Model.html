    public set(key: string | Object, value: string | Object, options: Object): boolean source

    Model set.

    Params:

    NameTypeAttributeDescription
    key string | Object

    key can be string or {}.
    value string | Object
    • optional

    value can be string or {}, not used if the key is an {}.
    options Object
    • optional

    model options.

    Return:

    boolean

    Example:

    // Set a model with a string key and string as value
model.set("name", "Vitalie")
    // Set a model with a string key and object as value
model.set("person", {
  firstName: "Vitalie",
  lastName: "Jerebnii"
})
    // Set a model with an object and mixed values
model.set({
  id: 10,
  person: {
    firstName: "Vitalie",
    lastName: "Jerebnii"
  }
})
    // Set a model without dispatching an event (silent mode)
model.set("name", "Vitalie", {silent: true})
