    // CircleButton.js
class TestText extends createjs.Container {

  constructor(text = '', color = '#222') {
    // invoke Container constructor
    super();
    // set props
    this.text = text;
    this.color = color;
    // Init component
    this.setup();
  }

  setup() {
    // Create a Text
    const txt = new createjs.Text(this.text, "20px Arial", this.color);
    this.addChild(txt);
  }
  // super.[method]
  draw(ctx, ignoreCache) {
    super.draw(ctx, ignoreCache)
    //console.log("Every time I")
  }
}

export default createjs.promote(TestText, "Container")
